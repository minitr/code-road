# 2319. 判断矩阵是否是一个 X 矩阵 (opens new window)

进行是否为对角线的判断即可

bool checkXMatrix(vector<vector<int>>& grid) {
    int n=grid.size();
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(i==j||i+j==n-1){
                if(!grid[i][j]) return false;
            }else if(grid[i][j]){
                return false;
            }
        }
    }
    return true;
}

# 599. 两个列表的最小索引总和 (opens new window)

  1. 直接暴力遍历
    vector<string> findRestaurant(vector<string>& list1, vector<string>& list2) {
        int minIndexSum=2001;
        int sum=0;
        vector<string> ans;
        for(int i=0;i<list1.size();++i){
            for(int j=0;j<list2.size();++j){
                if(list1[i]==list2[j]){
                    sum=i+j;
                    if(sum<minIndexSum){
                        minIndexSum=sum;
                        ans={list1[i]};
                    }else if(sum==minIndexSum){
                        ans.push_back(list1[i]);
                    }
                }
            }
        }
        return ans;
    }
  1. 使用map
     vector<string> findRestaurant(vector<string>& list1, vector<string>& list2) {
        int minIndexSum=2001;
        unordered_map<string, int> map;
        for(int i=0;i<list1.size();++i){
            map[list1[i]]=i;
        }
        vector<string> ans;
        for(int i=0;i<list2.size();++i){
            if(map.count(list2[i])){
                int indexSum=map[list2[i]]+i;
                if(indexSum<minIndexSum){
                    ans={list2[i]};
                    minIndexSum=indexSum;
                }else if(indexSum==minIndexSum){
                    ans.push_back(list2[i]);
                }
            }
        }
        return ans;
    }

# 674. 最长连续递增序列 (opens new window)

利用双指针判断递增序列

    int findLengthOfLCIS(vector<int>& nums) {
        if(nums.size()==1){
            return 1;
        }
        int begin=0;
        int end=1;
        int maxLength=0;
        while(end<nums.size()){
            while(end<nums.size()&&nums[end]>nums[end-1]){
                ++end;
            }
            int len=end-begin;
            if(len>maxLength){
                maxLength=len;
            }
            begin=end;
            end=begin+1;
        }
        return maxLength;
    }

动态规划

    int findLengthOfLCIS(vector<int>& nums) {
        if(nums.size()==1){
            return 1;
        }
        int dp[10001];
        dp[0]=1;
        int maxLength=0;
        for(int i=1;i<nums.size();++i){
            if(nums[i]>nums[i-1]){
                dp[i]=dp[i-1]+1;
            }else{
                dp[i]=1;
            }
            maxLength=max(maxLength,dp[i]);
        }
        return maxLength;
    }

# 989. 数组形式的整数加法 (opens new window)

注意判断条件是要判断两个下标是否超出对应数组长度，剩余的就是考虑进位加法了

    public List<Integer> addToArrayForm(int[] num, int k) {
        List<Integer> ans = new ArrayList<>();
        List<Integer> kList = new ArrayList<>();
        while (k > 0) {
            kList.add(k % 10);
            k /= 10;
        }
        int j = 0;
        int index = 0;
        int i = num.length - 1;
        while (i >= 0 || j < kList.size()) {
            int a = i >= 0 ? num[i] : 0;
            int b = j < kList.size() ? kList.get(j) : 0;
            int sum = a + b + index;
            ans.add(0, sum % 10);
            index = sum / 10;
            ++j;
            --i;
        }
        if (index != 0) {
            ans.add(0, index);
        }
        return ans;
    }
